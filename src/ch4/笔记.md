- 比较运算符
  - 用== 比较数组
    - 相同维数且相同个数元素的数组才可以比较
    - 每个元素都相等的才相等

~~~go
func TestComArray(t *testing.T){
  a := [...]int{1,2,3,4}
  b := [...]int{1,3,3,4}
  c := [...]int{1,2,3,4,5}
  d := [...]int{1,2,3,4}

  t.Log(a==b)
  // t.Log(a==c)
  t.Log(a==d)
}
~~~
